
CREATE MATERIALIZED VIEW "SYSTEM"."PARTIES"
("ID","USER_NAME","FIRST_NAME","LAST_NAME","STATUS","ACCOUNT_NUMBER","LEGAL_ENTITY_ID")
ORGANIZATION HEAP PCTFREE 10 PCTUSED 40
INITRANS 1 MAXTRANS 255
NOCOMPRESS LOGGING
STORAGE(INITIAL 65536 NEXT 1048576 MINEXTENTS 1
MAXEXTENTS 214748345
PCTINCREASE 0 FREELIST 1 FREELIST GROUPS 1
BUFFER_POOL DEFAULT FLASH_CACHE DEFAULT CELL_FLASH_CACHE DEFAULT)
BUILD IMMEDIATE
USING INDEX
REFRESH COMPLETE ON DEMAND START WITH SYSDATE +0 NEXT TRUNC(SYSDATE+1)+9/1440
USING DEFAULT LOCAL ROLLBACK SEGMENT
USING ENFORCED CONSTRAINTS DISABLE QUERY
REWRITE
AS
SELECT PP.ID,PP.USER_NAME,
PP.FIRST_NAME,
PP.LAST_NAME,PP.STATUS,PP.ACCOUNT_NUMBER,
PP.LEGAL_ENTITY_ID
FROM PARTIES PP;

COMMENT ON MATERIALIZED VIEW "SYSTEM"."PARTIES" IS 'SNAPSHOT TABLE FOR
PARTIES';
